let draggedIndex = null; function render(){ const el = document.getElementById("timeline"); el.innerHTML = ""; const s = slots(); data.forEach((e, i) => { const d = document.createElement("div"); d.className = "card"; d.draggable = true; d.dataset.index = i; d.innerHTML = ` ${s[i]}

${e.title}

${e.details}

`; d.onclick = () => showDetails(i); // ✅ Drag start d.ondragstart = () => { draggedIndex = i; d.classList.add("dragging"); }; d.ondragend = () => { d.classList.remove("dragging"); }; el.appendChild(d); }); // ✅ Handle drop on container (not card!) el.ondragover = (e) => { e.preventDefault(); }; el.ondrop = (e) => { e.preventDefault(); const afterElement = getDragAfterElement(el, e.clientX); const newIndex = afterElement ? parseInt(afterElement.dataset.index) : data.length; const item = data.splice(draggedIndex, 1)[0]; data.splice(newIndex, 0, item); render(); }; } function getDragAfterElement(container, x){ const elements = [...container.querySelectorAll(".card:not(.dragging)")]; return elements.reduce((closest, child) => { const box = child.getBoundingClientRect(); const offset = x - box.left - box.width / 2; if (offset < 0 && offset > closest.offset) { return { offset: offset, element: child }; } else { return closest; } }, { offset: Number.NEGATIVE_INFINITY }).element; }
